CREATE TABLE producto(id_prod int (15) AUTO_INCREMENT,
                      nombre_prod varchar(100) NOT null,
                      precio_prod int(25) NOT null,
		      descrp_prod varchar(100) NOT null,
                      des_det_prod varchar(500) NOT null,
PRIMARY KEY (id_prod))ENGINE =INNODB DEFAULT CHARSET=utf8 AUTO_INCREMENT=1;

CREATE TABLE rol(id_rol int (15) AUTO_INCREMENT,
                      cargo_rol varchar(100) NOT null,
PRIMARY KEY (id_rol))ENGINE =INNODB DEFAULT CHARSET=utf8 AUTO_INCREMENT=1;

CREATE TABLE usuario(id_usuario int (15) AUTO_INCREMENT,
                     nombre_us varchar(100) NOT null,
		     cuenta_us varchar(100) NOT null,
		     correo_us varchar(100) NOT null,
                     contrase√±a_us varchar(100) NOT null,
		     telefono_us varchar(15) NOT null,
		     id_tipo int(15) not null,
FOREIGN KEY (id_tipo) REFERENCES rol(id_rol),
PRIMARY KEY (id_usuario))ENGINE =INNODB DEFAULT CHARSET=utf8 AUTO_INCREMENT=1;

CREATE TABLE registrados(id_regis int (15) AUTO_INCREMENT,
                         fk_rol int (15),
                         fk_usuario int (15),
FOREIGN KEY (fk_rol) REFERENCES rol(id_rol),
FOREIGN KEY (fk_usuario) REFERENCES usuario(id_usuario),
PRIMARY KEY (id_regis))ENGINE =INNODB DEFAULT CHARSET=utf8 AUTO_INCREMENT=1;


CREATE TABLE sav_regis(id_sav int (15) AUTO_INCREMENT,
    			 sav_id_regis int (15),
                         sav_fk_rol int (15),
                         sav_fk_usuario int (15),
			 sav_us_nombre varchar(100),
			 sav_us_cuenta varchar(100),
			 sav_us_correo varchar(100),
			 sav_us_telefono varchar(100),
FOREIGN KEY (sav_fk_rol) REFERENCES rol(id_rol),
FOREIGN KEY (sav_fk_usuario) REFERENCES usuario(id_usuario),
PRIMARY KEY (id_sav))ENGINE =INNODB DEFAULT CHARSET=utf8 AUTO_INCREMENT=1;
